                    return strNum;
                }
                if(!/^(\+|-)?(\d+)(\.\d+)?$/.test(strNum)) {
                    return strNum;
                }
                var a = RegExp.$1,
                    b = RegExp.$2,
                    c = RegExp.$3;
                var re = new RegExp();
                re.compile("(\\d)(\\d{3})(,|$)");
                while(re.test(b)) {
                    b = b.replace(re, "$1,$2$3");
                }
                return a + "" + b + "" + c;
            }

		// 就是判断是否为整数(正负)
		Pattern pattern = 
			Pattern.compile("^\\d+$|-\\d+$"); 
		//判断是否为小数(正负)
		Pattern pattern2 = 
			Pattern.compile("\\d+\\.\\d+$|-\\d+\\.\\d+$");
		return (pattern.matcher(str).matches() 
			|| pattern2.matcher(str).matches());
	}
